```typescript
// packages/api/src/features/example/repository.ts
import { eq } from 'drizzle-orm';
import type { Database } from '@/db/index.ts';
import { db as defaultDb } from '@/db/index.ts';
import { type ExampleRecord, examples, type NewExampleRecord } from './schema.ts';

export function createExampleRepository(db: Database) {
  return {
    async create(data: NewExampleRecord): Promise<ExampleRecord> {
      const [record] = await db.insert(examples).values(data).returning();
      if (!record) throw new Error('Failed to create record');
      return record;
    },

    async findById(id: string): Promise<ExampleRecord | null> {
      const [record] = await db.select().from(examples).where(eq(examples.id, id));
      return record ?? null;
    },

    async findByDate(date: string): Promise<ExampleRecord | null> {
      const [record] = await db
        .select()
        .from(examples)
        .where(eq(examples.date, date));
      return record ?? null;
    },

    async update(
      id: string,
      data: Partial<NewExampleRecord>,
    ): Promise<ExampleRecord | null> {
      const [record] = await db
        .update(examples)
        .set({ ...data, updatedAt: new Date() })
        .where(eq(examples.id, id))
        .returning();
      return record ?? null;
    },

    async delete(id: string): Promise<boolean> {
      const result = await db
        .delete(examples)
        .where(eq(examples.id, id))
        .returning({ id: examples.id });
      return result.length > 0;
    },
  };
}

// デフォルトインスタンス（本番用）
export const ExampleRepository = createExampleRepository(defaultDb);
```

```typescript
// packages/api/src/features/example/service.ts
import type { CreateExample } from '@app/shared';
import { ConflictError, NotFoundError } from '@/lib/errors.ts';
import { ExampleRepository, createExampleRepository } from './repository.ts';

export function createExampleService(
  repo: ReturnType<typeof createExampleRepository>,
) {
  return {
    async create(data: CreateExample) {
      const existing = await repo.findByDate(data.date);
      if (existing) throw new ConflictError('この日付のデータは既に存在します');
      return repo.create(data);
    },

    async getById(id: string) {
      const record = await repo.findById(id);
      if (!record) throw new NotFoundError('データが見つかりません');
      return record;
    },

    async update(id: string, data: Partial<CreateExample>) {
      const record = await repo.update(id, data);
      if (!record) throw new NotFoundError('データが見つかりません');
      return record;
    },

    async delete(id: string) {
      const deleted = await repo.delete(id);
      if (!deleted) throw new NotFoundError('データが見つかりません');
    },
  };
}

// デフォルトインスタンス（本番用）
export const ExampleService = createExampleService(ExampleRepository);
```

ルート定義（`createRoute()` の戻り値）はファクトリの外に維持。ハンドラ登録部分のみファクトリ内に移動。

```typescript
// packages/api/src/features/example/routes.ts
import { createRoute, OpenAPIHono } from '@hono/zod-openapi';
import { AppError } from '@/lib/errors.ts';
import {
  CreateExampleRequestSchema,
  ErrorResponseSchema,
  ExampleResponseSchema,
  IdParamSchema,
  UpdateExampleRequestSchema,
} from './openapi.ts';
import { ExampleService, createExampleService } from './service.ts';
import type { ExampleRecord } from './schema.ts';

function formatResponse(record: ExampleRecord) {
  return {
    id: record.id,
    date: record.date,
    field1: record.field1,
    field2: record.field2,
    createdAt: record.createdAt.toISOString(),
    updatedAt: record.updatedAt.toISOString(),
  };
}

// --- ルート定義（ファクトリの外：サービスに依存しない） ---

const createExampleRoute = createRoute({
  method: 'post',
  path: '/',
  tags: ['Example'],
  summary: 'データを作成',
  request: {
    body: {
      content: { 'application/json': { schema: CreateExampleRequestSchema } },
      required: true,
    },
  },
  responses: {
    201: {
      content: { 'application/json': { schema: ExampleResponseSchema } },
      description: '作成成功',
    },
    409: {
      content: { 'application/json': { schema: ErrorResponseSchema } },
      description: '重複エラー',
    },
  },
});

const getExampleRoute = createRoute({
  method: 'get',
  path: '/{id}',
  tags: ['Example'],
  summary: 'データを取得',
  request: { params: IdParamSchema },
  responses: {
    200: {
      content: { 'application/json': { schema: ExampleResponseSchema } },
      description: '取得成功',
    },
    404: {
      content: { 'application/json': { schema: ErrorResponseSchema } },
      description: 'Not Found',
    },
  },
});

const updateExampleRoute = createRoute({
  method: 'patch',
  path: '/{id}',
  tags: ['Example'],
  summary: 'データを更新',
  request: {
    params: IdParamSchema,
    body: {
      content: { 'application/json': { schema: UpdateExampleRequestSchema } },
      required: true,
    },
  },
  responses: {
    200: {
      content: { 'application/json': { schema: ExampleResponseSchema } },
      description: '更新成功',
    },
    404: {
      content: { 'application/json': { schema: ErrorResponseSchema } },
      description: 'Not Found',
    },
  },
});

const deleteExampleRoute = createRoute({
  method: 'delete',
  path: '/{id}',
  tags: ['Example'],
  summary: 'データを削除',
  request: { params: IdParamSchema },
  responses: {
    204: { description: '削除成功' },
    404: {
      content: { 'application/json': { schema: ErrorResponseSchema } },
      description: 'Not Found',
    },
  },
});

// --- ファクトリ関数 ---

export function createExampleRoutes(
  service: ReturnType<typeof createExampleService>,
) {
  const baseApp = new OpenAPIHono();

  baseApp.onError((err, c) => {
    if (err instanceof AppError) {
      return c.json(
        { error: { message: err.message } },
        err.statusCode as 400 | 404 | 409 | 500,
      );
    }
    throw err;
  });

  return baseApp
    .openapi(createExampleRoute, async (c) => {
      const data = c.req.valid('json');
      const record = await service.create(data);
      return c.json(formatResponse(record), 201);
    })
    .openapi(getExampleRoute, async (c) => {
      const { id } = c.req.valid('param');
      const record = await service.getById(id);
      return c.json(formatResponse(record), 200);
    })
    .openapi(updateExampleRoute, async (c) => {
      const { id } = c.req.valid('param');
      const data = c.req.valid('json');
      const record = await service.update(id, data);
      return c.json(formatResponse(record), 200);
    })
    .openapi(deleteExampleRoute, async (c) => {
      const { id } = c.req.valid('param');
      await service.delete(id);
      return c.body(null, 204);
    });
}

// デフォルトインスタンス（本番用）
export const exampleRoutes = createExampleRoutes(ExampleService);
```

```typescript
      expect(found!.field2).toBe('test');
    });

    it('存在しないIDでnullを返す', async () => {
      const found = await repo.findById('00000000-0000-0000-0000-000000000000');
      expect(found).toBeNull();
    });
  });

  describe('findByDate', () => {
    it('日付でレコードを取得できる', async () => {
      await repo.create({ date: '2024-01-15', field2: 'test' });
      const found = await repo.findByDate('2024-01-15');

        { field2: 'x' },

    });
  });

  describe('delete', () => {

      const deleted = await repo.delete(created.id);
      expect(deleted).toBe(true);

    it('存在しないIDでfalseを返す', async () => {
```

```typescript
// packages/api/src/features/example/__tests__/service.test.ts
import { describe, expect, it } from '@/test/setup.ts';
import { ConflictError, NotFoundError } from '@/lib/errors.ts';
import { createExampleService } from '../service.ts';
import { createExampleRepository } from '../repository.ts';
import type { ExampleRecord } from '../schema.ts';

function mockRecord(overrides: Partial<ExampleRecord> = {}): ExampleRecord {
  return {
    id: '00000000-0000-0000-0000-000000000001',
    date: '2024-01-15',
    field1: false,
    field2: 'test',
    createdAt: new Date('2024-01-15T00:00:00Z'),
    updatedAt: new Date('2024-01-15T00:00:00Z'),
    ...overrides,
  };
}

type MockRepo = ReturnType<typeof createExampleRepository>;

function createMockRepo(overrides: Partial<MockRepo> = {}): MockRepo {
  return {
    create: async (data) => mockRecord(data),
    findById: async () => null,
    findByDate: async () => null,
    update: async () => null,
    delete: async () => false,
    ...overrides,
  } as MockRepo;
}

describe('ExampleService', () => {
  describe('create', () => {
    it('新しいデータを作成できる', async () => {
      const repo = createMockRepo();
      const service = createExampleService(repo);

      const record = await service.create({
        date: '2024-01-15',
        field1: true,
        field2: 'test',
      });
      expect(record.date).toBe('2024-01-15');
    });

    it('同一日付が存在する場合ConflictErrorを投げる', async () => {
      const repo = createMockRepo({
        findByDate: async () => mockRecord(),
      });
      const service = createExampleService(repo);

      await expect(
        service.create({ date: '2024-01-15', field1: false, field2: 'test' }),
      ).rejects.toThrow(ConflictError);
    });
  });

  describe('getById', () => {
    it('IDでデータを取得できる', async () => {
      const repo = createMockRepo({
        findById: async () => mockRecord({ id: 'abc' }),
      });
      const service = createExampleService(repo);

      const record = await service.getById('abc');
      expect(record.id).toBe('abc');
    });

    it('存在しない場合NotFoundErrorを投げる', async () => {
      const repo = createMockRepo();
      const service = createExampleService(repo);

      await expect(service.getById('nonexistent')).rejects.toThrow(
        NotFoundError,
      );
    });
  });

  describe('update', () => {
    it('データを更新できる', async () => {
      const repo = createMockRepo({
        update: async (_id, data) => mockRecord({ ...data }),
      });
      const service = createExampleService(repo);

      const record = await service.update('id', { field2: 'updated' });
      expect(record.field2).toBe('updated');
    });

    it('存在しない場合NotFoundErrorを投げる', async () => {
      const repo = createMockRepo();
      const service = createExampleService(repo);

      await expect(
        service.update('nonexistent', { field2: 'x' }),
      ).rejects.toThrow(NotFoundError);
    });
  });

  describe('delete', () => {
    it('データを削除できる', async () => {
      const repo = createMockRepo({
        delete: async () => true,
      });
      const service = createExampleService(repo);

      // 例外が投げられなければ成功
      await service.delete('id');
    });

    it('存在しない場合NotFoundErrorを投げる', async () => {
      const repo = createMockRepo();
```

## Task 6: Routes 統合テストを実装

**Files:**
- Create: `packages/api/src/features/example/__tests__/routes.test.ts`

**Step 1: テストファイルを作成**

ルートの統合テストはPGLiteを使い、ファクトリでテスト用アプリを構築してHTTPサイクルを検証する。

```typescript
// packages/api/src/features/example/__tests__/routes.test.ts
import {
  afterAll,
  afterEach,
  beforeAll,
  cleanupTables,
  closePglite,
  describe,
  expect,
  it,
  setupTestDb,
} from '@/test/setup.ts';
import { OpenAPIHono } from '@hono/zod-openapi';
import type { Database } from '@/db/index.ts';
import { createExampleRepository } from '../repository.ts';
import { createExampleService } from '../service.ts';
import { createExampleRoutes } from '../routes.ts';

describe('Example Routes', () => {
  let db: Database;
  let app: InstanceType<typeof OpenAPIHono>;

  beforeAll(async () => {
    db = await setupTestDb();
    const repo = createExampleRepository(db);
    const service = createExampleService(repo);
    const routes = createExampleRoutes(service);
    app = new OpenAPIHono().route('/api/example', routes);
  });

  afterEach(async () => {
    await cleanupTables(db);
  });

  afterAll(async () => {
    await closePglite();
  });

  // --- ヘルパー ---
  function post(body: Record<string, unknown>) {
    return app.request('/api/example', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
  }

  function get(id: string) {
    return app.request(`/api/example/${id}`);
  }

  function patch(id: string, body: Record<string, unknown>) {
    return app.request(`/api/example/${id}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
  }

  function del(id: string) {
    return app.request(`/api/example/${id}`, { method: 'DELETE' });
  }

  // --- POST ---
  describe('POST /api/example', () => {
    it('201: データを作成できる', async () => {
      const res = await post({ date: '2024-01-15', field1: true, field2: 'test' });
      expect(res.status).toBe(201);
      const body = await res.json();
      expect(body.id).toBeDefined();
      expect(body.date).toBe('2024-01-15');
      expect(body.field1).toBe(true);
      expect(body.field2).toBe('test');
      expect(body.createdAt).toBeDefined();
      expect(body.updatedAt).toBeDefined();
    });

    it('201: field1はデフォルトfalse', async () => {
      const res = await post({ date: '2024-01-15', field2: 'test' });
      expect(res.status).toBe(201);
      const body = await res.json();
      expect(body.field1).toBe(false);
    });

    it('409: 同一日付で重複エラー', async () => {
      await post({ date: '2024-01-15', field2: 'first' });
      const res = await post({ date: '2024-01-15', field2: 'second' });
      expect(res.status).toBe(409);
      const body = await res.json();
      expect(body.error.message).toBeDefined();
    });

    it('400: field2が欠けている場合バリデーションエラー', async () => {
      const res = await post({ date: '2024-01-15' });
      expect(res.status).toBe(400);
    });

    it('400: 不正な日付形式', async () => {
      const res = await post({ date: 'not-a-date', field2: 'test' });
      expect(res.status).toBe(400);
    });
  });

  // --- GET ---
  describe('GET /api/example/:id', () => {
    it('200: IDでデータを取得できる', async () => {
      const createRes = await post({ date: '2024-01-15', field2: 'test' });
      const { id } = await createRes.json();

      const res = await get(id);
      expect(res.status).toBe(200);
      const body = await res.json();
      expect(body.id).toBe(id);
    });

    it('404: 存在しないID', async () => {
      const res = await get('00000000-0000-0000-0000-000000000000');
      expect(res.status).toBe(404);
    });

    it('400: 不正なUUID形式', async () => {
      const res = await get('not-a-uuid');
      expect(res.status).toBe(400);
    });
  });

  // --- PATCH ---
  describe('PATCH /api/example/:id', () => {
    it('200: データを更新できる', async () => {
      const createRes = await post({ date: '2024-01-15', field2: 'original' });
      const { id } = await createRes.json();

      const res = await patch(id, { field2: 'updated' });
      expect(res.status).toBe(200);
      const body = await res.json();
      expect(body.field2).toBe('updated');
    });

    it('200: 一部フィールドのみ更新', async () => {
      const createRes = await post({
        date: '2024-01-15',
        field1: false,
        field2: 'original',
      });
      const { id } = await createRes.json();

      const res = await patch(id, { field1: true });
      expect(res.status).toBe(200);
      const body = await res.json();
      expect(body.field1).toBe(true);
      expect(body.field2).toBe('original'); // 変更なし
    });

    it('404: 存在しないID', async () => {
      const res = await patch('00000000-0000-0000-0000-000000000000', {

      });
      expect(res.status).toBe(404);
    });
  });

  // --- DELETE ---
  describe('DELETE /api/example/:id', () => {
    it('204: データを削除できる', async () => {

      const { id } = await createRes.json();

      const res = await del(id);
      expect(res.status).toBe(204);

      // 削除後はGETで404
      const getRes = await get(id);
      expect(getRes.status).toBe(404);
    });

    it('404: 存在しないID', async () => {
      const res = await del('00000000-0000-0000-0000-000000000000');
      expect(res.status).toBe(404);
    });
  });
```
